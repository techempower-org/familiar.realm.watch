# familiar v0.2 — "The familiar remembers better"

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Harden familiar with local llama.cpp inference on katana, multi-endpoint failover with circuit breakers, Emmimal retrieval components 2–4 (rerank, decay, extractive compression), a confidence gate + stuck detector, `/api/familiar/eval`, an MCP server, diary buffering, and PWA citation popovers.

**Architecture:** familiar-api gains a `LlamaCppClient` (OpenAI-compat HTTP to katana's `llama-server`) and an `InferenceRouter` that tries endpoints in priority order — katana llama.cpp first, familiar's Ollama second — wrapping each in the existing `CircuitBreaker`. The retrieval pipeline in `memory-protocol.ts` gains three post-search steps: domain-weighted rerank, exponential temporal decay, and extractive compression (Emmimal components 2, 3, 4). A diary buffer accumulates post-stream writes and flushes every 10 turns. The MCP server exposes `familiar_chat`, `familiar_recall`, and `familiar_reflect` via `@modelcontextprotocol/sdk` over HTTP. The PWA replaces `[drawer_xxx]` citation markers with hover popovers built entirely from DOM methods.

**Tech Stack:** TypeScript + Bun, `@modelcontextprotocol/sdk` (^1.29, already installed), llama.cpp (cmake build with CUDA on katana), `bun:test` for all unit tests.

**Key constraints:**
- Bun, not Node — `bun:test`, `Bun.spawnSync({ cmd: [...] })`, native TS.
- No `innerHTML` anywhere in web JS — use `document.createElement` + `textContent`.
- No subprocess shell strings — use argv arrays with `Bun.spawnSync`.
- Emmimal retrieval components are client-side TypeScript in `src/retrieval/`, applied to results returned by palace-daemon.

```typescript
import type { ChatStreamOpts } from "./ollama-client.ts";
import type { OllamaChatChunk } from "./types.ts";

export interface LlamaCppClientOptions {
  baseUrl: string;
  model: string;
  fetch?: typeof fetch;
}

/**
 * HTTP client for llama.cpp's OpenAI-compatible server (llama-server).
 * Emits OllamaChatChunk-shaped objects so it can substitute for OllamaClient
 * in InferenceRouter without changes to chat.ts.
 */
export class LlamaCppClient {
  private baseUrl: string;
  private model: string;
  private fetchFn: typeof fetch;

  constructor(opts: LlamaCppClientOptions) {
    this.baseUrl = opts.baseUrl.replace(/\/$/, "");
    this.model = opts.model;
    this.fetchFn = opts.fetch ?? fetch;
  }

  async isHealthy(): Promise<boolean> {
    try {
      const res = await this.fetchFn(`${this.baseUrl}/health`);
      return res.ok;
    } catch {
      return false;
    }
  }

  async *chatStream(opts: ChatStreamOpts): AsyncGenerator<OllamaChatChunk> {
    const body = {
      model: opts.model ?? this.model,
      messages: opts.messages,
      stream: true,
      options: opts.options,
    };
    const res = await this.fetchFn(`${this.baseUrl}/v1/chat/completions`, {
      method: "POST",
      headers: { "content-type": "application/json" },
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error(`llama-cpp chat: ${res.status} ${res.statusText}`);
    if (!res.body) throw new Error("llama-cpp chat: no response body");

    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    let buffer = "";

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });

      let newlineIdx: number;
      while ((newlineIdx = buffer.indexOf("\n")) >= 0) {
        const line = buffer.slice(0, newlineIdx).trim();
        buffer = buffer.slice(newlineIdx + 1);
        if (!line.startsWith("data:")) continue;
        const payload = line.slice(5).trim();
        if (payload === "[DONE]") {
          yield { model: this.model, created_at: new Date().toISOString(), done: true };
          return;
        }
        try {
          const parsed = JSON.parse(payload);
          const delta = parsed.choices?.[0]?.delta?.content ?? "";
          const isDone = parsed.choices?.[0]?.finish_reason === "stop";
          yield {
            model: this.model,
            created_at: new Date().toISOString(),
            message: delta ? { role: "assistant", content: delta } : undefined,
            done: isDone,
          } as OllamaChatChunk;
        } catch { /* skip malformed SSE lines */ }
      }
    }
    yield { model: this.model, created_at: new Date().toISOString(), done: true };
  }
}
```

```typescript
import { CircuitBreaker, type CircuitBreakerOptions } from "./circuit-breaker.ts";
import type { ChatStreamOpts } from "./ollama-client.ts";
import type { InferenceChatProvider, OllamaChatChunk } from "./types.ts";

const DEFAULT_BREAKER: CircuitBreakerOptions = {
  threshold: 3,
  windowMs: 30_000,
  openMs: 60_000,
};

export class InferenceRouter implements InferenceChatProvider {
  private providers: InferenceChatProvider[];
  private breakers: CircuitBreaker[];

  constructor(providers: InferenceChatProvider[], breakerOpts?: Partial<CircuitBreakerOptions>) {
    this.providers = providers;
    const opts = { ...DEFAULT_BREAKER, ...breakerOpts };
    this.breakers = providers.map(() => new CircuitBreaker(opts));
  }

  /** Expose breaker by index for testing. */
  breakerFor(i: number): CircuitBreaker {
    return this.breakers[i];
  }

  async isHealthy(): Promise<boolean> {
    for (const p of this.providers) {
      if (await p.isHealthy()) return true;
    }
    return false;
  }

  async *chatStream(opts: ChatStreamOpts): AsyncGenerator<OllamaChatChunk> {
    for (let i = 0; i < this.providers.length; i++) {
      const breaker = this.breakers[i];
      if (breaker.state() === "open") continue;
      try {
        yield* await this._streamWithBreaker(this.providers[i], breaker, opts);
        return;
      } catch {
        // breaker already recorded the failure; try next
      }
    }
    throw new Error("all inference endpoints failed or circuit-broken");
  }

  private async _streamWithBreaker(
    provider: InferenceChatProvider,
    breaker: CircuitBreaker,
    opts: ChatStreamOpts
  ): Promise<AsyncGenerator<OllamaChatChunk>> {
    // We need to wrap the generator inside breaker.run() so failures record.
    // Strategy: eagerly pull the first value inside breaker.run; if it throws,
    // the breaker records it. Then yield buffered + remainder outside.
    const gen = provider.chatStream(opts);
    const buffer: OllamaChatChunk[] = [];

    await breaker.run(async () => {
      const first = await gen.next();
      if (!first.done) buffer.push(first.value);
    });

    async function* emit(): AsyncGenerator<OllamaChatChunk> {
      yield* buffer;
      for await (const chunk of gen) yield chunk;
    }
    return emit();
  }
}
```

Create `src/retrieval/rerank.ts`:

```typescript
import type { PalaceDrawer } from "../types.ts";

const WING_WEIGHT = 0.68;
const TAG_WEIGHT = 0.32;
const WING_BOOST = 1.4;
const RECENCY_BONUS = 0.1;
const RECENCY_WINDOW_MS = 48 * 3600 * 1000;

function tagImportance(drawer: PalaceDrawer, wingScope: string | null): number {
  if (wingScope && drawer.wing === wingScope) return WING_BOOST;
  return 1.0;
}

function recencyBonus(drawer: PalaceDrawer): number {
  if (!drawer.created_at) return 0;
  try {
    const age = Date.now() - new Date(drawer.created_at).getTime();
    return age <= RECENCY_WINDOW_MS ? RECENCY_BONUS : 0;
  } catch {
    return 0;
  }
}

function finalScore(drawer: PalaceDrawer, wingScope: string | null): number {
  const base = drawer.similarity ?? drawer.distance ?? 0;
  return base * WING_WEIGHT + tagImportance(drawer, wingScope) * TAG_WEIGHT + recencyBonus(drawer);
}

/**
 * Rerank drawers by domain-weighted score.
 * Returns a new array sorted descending; mutates similarity field with final score.
 */
export function domainRerank(drawers: PalaceDrawer[], wingScope: string | null): PalaceDrawer[] {
  return drawers
    .map((d) => ({ ...d, similarity: finalScore(d, wingScope) }))
    .sort((a, b) => (b.similarity ?? 0) - (a.similarity ?? 0));
}
```

```typescript
import type { PalaceDrawer } from "../types.ts";

export interface DecayOptions {
  halfLifeDays: number;
}

export function temporalDecay(drawers: PalaceDrawer[], opts: DecayOptions): PalaceDrawer[] {
  const lambda = Math.log(2) / opts.halfLifeDays;
  const now = Date.now();

  return drawers.map((d) => {
    const base = d.similarity ?? d.distance ?? 0;
    if (!d.created_at) return { ...d, similarity: base };
    try {
      const ageDays = (now - new Date(d.created_at).getTime()) / 86400_000;
      const decayed = base * Math.exp(-lambda * ageDays);
      return { ...d, similarity: decayed };
    } catch {
      return { ...d, similarity: base };
    }
  });
}
```

```typescript
import type { PalaceDrawer } from "../types.ts";

const SHORT_THRESHOLD = 500;
const MAX_SENTENCES = 3;

function tokenOverlap(a: string, b: string): number {
  const aWords = new Set(a.toLowerCase().match(/\b\w+\b/g) ?? []);
  const bWords = new Set(b.toLowerCase().match(/\b\w+\b/g) ?? []);
  let hits = 0;
  for (const w of aWords) if (bWords.has(w)) hits++;
  return hits / Math.max(aWords.size, 1);
}

function splitSentences(text: string): string[] {
  return text.split(/(?<=[.!?])\s+/).filter((s) => s.trim().length > 0);
}

function compressDrawer(drawer: PalaceDrawer, query: string): PalaceDrawer {
  if (drawer.text.length <= SHORT_THRESHOLD) return drawer;

  const sentences = splitSentences(drawer.text);
  if (sentences.length <= MAX_SENTENCES) return drawer;

  const scored = sentences.map((s, idx) => ({ s, score: tokenOverlap(s, query), idx }));
  const top = scored
    .sort((a, b) => b.score - a.score)
    .slice(0, MAX_SENTENCES)
    .sort((a, b) => a.idx - b.idx) // restore original order
    .map((x) => x.s);

  return { ...drawer, text: top.join(" ") };
}

export function extractiveCompress(drawers: PalaceDrawer[], query: string): PalaceDrawer[] {
  return drawers.map((d) => compressDrawer(d, query));
}
```

```typescript
import type { PalaceClient } from "../palace-client.ts";
import type { Config } from "../types.ts";
import { retrieveAndGround } from "../memory-protocol.ts";

export interface EvalRouteDeps {
  cfg: Config;
  palace: PalaceClient;
}

interface EvalRequest {
  prompt: string;
  expected_keywords?: string[];
  mock_response?: string;
  wing?: string;
}

interface EvalResponse {
  prompt: string;
  response: string;
  relevance_score: number;
  drawer_count: number;
  drawer_ids: string[];
  warnings: string[];
  faithfulness: boolean;
}

export function scoreEvalResult(response: string, keywords: string[]): number {
  if (!keywords.length) return 1.0;
  const lower = response.toLowerCase();
  const hits = keywords.filter((kw) => lower.includes(kw.toLowerCase())).length;
  return hits / keywords.length;
}

export async function handleEval(req: Request, deps: EvalRouteDeps): Promise<Response> {
  let body: EvalRequest;
  try {
    body = (await req.json()) as EvalRequest;
  } catch {
    return new Response(JSON.stringify({ error: "invalid JSON body" }), {
      status: 400,
      headers: { "content-type": "application/json" },
    });
  }

  if (!body.prompt) {
    return new Response(JSON.stringify({ error: "prompt required" }), {
      status: 400,
      headers: { "content-type": "application/json" },
    });
  }

  let grounded: Awaited<ReturnType<typeof retrieveAndGround>>;
  try {
    grounded = await retrieveAndGround({
      palace: deps.palace,
      userMessage: body.prompt,
      wingScope: body.wing ?? null,
      retrievalLimit: deps.cfg.retrievalLimit,
      contextBudgetTokens: deps.cfg.tokenBudget.context,
      recentCitations: [],
    });
  } catch {
    grounded = { systemPrompt: "", drawerIds: [], warnings: ["palace_unreachable"] };
  }

  const response = body.mock_response ?? "(no mock_response provided — use for CI only)";
  const keywords = body.expected_keywords ?? [];
  const score = scoreEvalResult(response, keywords);
  const faithfulness = grounded.drawerIds.some((id) => response.includes(id));

  const result: EvalResponse = {
    prompt: body.prompt,
    response,
    relevance_score: score,
    drawer_count: grounded.drawerIds.length,
    drawer_ids: grounded.drawerIds,
    warnings: grounded.warnings,
    faithfulness,
  };

  return new Response(JSON.stringify(result, null, 2), {
    status: 200,
    headers: { "content-type": "application/json" },
  });
}
```

Create `src/diary-buffer.ts`:

```typescript
type FlushFn = (entries: string[]) => Promise<void>;

export interface DiaryBufferOptions {
  flushSize: number;
  flushFn: FlushFn;
}

export class DiaryBuffer {
  private entries: string[] = [];
  private flushFn: FlushFn;
  private flushSize: number;
  private flushing = false;

  constructor(opts: DiaryBufferOptions) {
    this.flushSize = opts.flushSize;
    this.flushFn = opts.flushFn;
  }

  size(): number { return this.entries.length; }

  async add(entry: string): Promise<void> {
    this.entries.push(entry);
    if (this.entries.length >= this.flushSize) {
      await this.flush();
    }
  }

  async flush(): Promise<void> {
    if (this.entries.length === 0 || this.flushing) return;
    this.flushing = true;
    const batch = this.entries.splice(0);
    try {
      await this.flushFn(batch);
    } finally {
      this.flushing = false;
    }
  }
}
```

```javascript
// Citation rendering — DOM-only, no innerHTML.

const CITATION_PATTERN = /\[drawer_([a-z0-9]+)\]/g;

function buildPopover(drawerId, meta) {
  const popover = document.createElement("span");
  popover.className = "citation-popover";
  popover.setAttribute("role", "tooltip");

  const title = document.createElement("strong");
  title.textContent = meta ? `${meta.wing} · ${meta.room}` : drawerId;
  popover.appendChild(title);

  if (meta?.created_at) {
    const date = document.createElement("span");
    date.className = "citation-date";
    date.textContent = new Date(meta.created_at).toLocaleDateString();
    popover.appendChild(date);
  }

  const link = document.createElement("a");
  link.className = "citation-link";
  link.href = `/api/familiar/memory/${drawerId}`;
  link.target = "_blank";
  link.rel = "noopener noreferrer";
  link.textContent = "View in palace →";
  popover.appendChild(link);

  return popover;
}

function buildCitationButton(rawId, meta) {
  const drawerId = "drawer_" + rawId;
  const wrapper = document.createElement("span");
  wrapper.className = "citation";
  wrapper.setAttribute("data-drawer-id", drawerId);

  const btn = document.createElement("button");
  btn.className = "citation-trigger";
  btn.setAttribute("aria-label", `Palace source: ${drawerId}`);
  btn.textContent = `[${rawId.slice(0, 6)}]`;

  const popover = buildPopover(drawerId, meta);
  wrapper.appendChild(btn);
  wrapper.appendChild(popover);

  btn.addEventListener("click", (e) => {
    e.stopPropagation();
    const isOpen = wrapper.classList.toggle("citation--open");
    if (isOpen && !meta) {
      fetch(`/api/familiar/memory/${drawerId}`)
        .then((r) => r.ok ? r.json() : null)
        .then((data) => {
          if (data) {
            const fresh = buildPopover(drawerId, data);
            wrapper.replaceChild(fresh, wrapper.querySelector(".citation-popover"));
          }
        })
        .catch(() => {});
    }
  });

  return wrapper;
}

/**
 * Render text containing [drawer_xxx] markers into a container.
 * Uses DOM text nodes and elements — never innerHTML.
 */
export function renderWithCitations(container, text, knownMeta = {}) {
  // Clear existing children
  while (container.firstChild) container.removeChild(container.firstChild);

  let lastIndex = 0;
  CITATION_PATTERN.lastIndex = 0;
  let match;

  while ((match = CITATION_PATTERN.exec(text)) !== null) {
    if (match.index > lastIndex) {
      container.appendChild(document.createTextNode(text.slice(lastIndex, match.index)));
    }
    const rawId = match[1];
    const meta = knownMeta[`drawer_${rawId}`] ?? null;
    container.appendChild(buildCitationButton(rawId, meta));
    lastIndex = match.index + match[0].length;
  }

  if (lastIndex < text.length) {
    container.appendChild(document.createTextNode(text.slice(lastIndex)));
  }
}
```
